## Smoke SQL

```sql
-- 1) Títulos com barra ou enumeração ruim
SELECT locale, COUNT(*)
FROM question_translations
WHERE locale = 'pt_BR'
  AND title LIKE '%/%'
GROUP BY locale;

-- 2) Perguntas começando em tom imperativo
SELECT locale, COUNT(*)
FROM question_translations
WHERE locale = 'pt_BR'
  AND (title ILIKE 'Informe %' OR title ILIKE 'Selecione %' OR title ILIKE 'Descreva %')
GROUP BY locale;

-- 3) Labels com barra em pt-BR
SELECT locale, COUNT(*)
FROM option_translations
WHERE locale = 'pt_BR'
  AND label LIKE '%/%'
GROUP BY locale;

-- 4) Labels acima de 50 caracteres em pt-BR
SELECT COUNT(*)
FROM option_translations
WHERE locale = 'pt_BR'
  AND LENGTH(label) > 50;

-- 5) Títulos acima de 80 caracteres em pt-BR
SELECT COUNT(*)
FROM question_translations
WHERE locale = 'pt_BR'
  AND LENGTH(title) > 80;

-- 6) Perguntas ainda em tom pouco conversacional
SELECT q.code, qt.title
FROM question_translations qt
JOIN questions q ON q.id = qt.question_id
WHERE qt.locale = 'pt_BR'
  AND qt.title NOT ILIKE 'Qual %'
  AND qt.title NOT ILIKE 'Como %'
  AND qt.title NOT ILIKE 'Que %'
  AND qt.title NOT ILIKE 'Quando %'
  AND qt.title NOT ILIKE 'Em quais %'
ORDER BY q.code;
```
